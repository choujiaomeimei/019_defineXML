## 4. 状态管理

### 4.1 用户状态 (User Store)
```typescript
export const useUserStore = defineStore('user', () => {
  const userInfo = ref<UserInfo | null>(null)
  const token = ref('')

  const isLoggedIn = computed(() => !!token.value)
  const username = computed(() => userInfo.value?.username || '')

  const login = async (credentials: LoginForm) => {
    const res = await ApiLogin(credentials)
    if (res.data.success) {
      token.value = res.data.token
      userInfo.value = res.data.user
      localStorage.setItem('token', token.value)
      localStorage.setItem('user', JSON.stringify(userInfo.value))
    }
  }

  const logout = () => {
    token.value = ''
    userInfo.value = null
    localStorage.removeItem('token')
    localStorage.removeItem('user')
  }

  const initFromStorage = () => {
    const storedToken = localStorage.getItem('token')
    const storedUser = localStorage.getItem('user')

    if (storedToken && storedUser) {
      token.value = storedToken
      userInfo.value = JSON.parse(storedUser)
    }
  }

  return {
    userInfo,
    token,
    isLoggedIn,
    username,
    login,
    logout,
    initFromStorage
  }
})
```

### 4.2 项目状态 (Project Store)
```typescript
export const useProjectStore = defineStore('project', () => {
  const currentProject = ref<ProjectDTO | null>(null)
  const projectList = ref<ProjectDTO[]>([])
  const loading = ref(false)

  const setCurrentProject = (project: ProjectDTO) => {
    currentProject.value = project
  }

  const fetchProjects = async (username: string) => {
    loading.value = true
    try {
      const res = await ApiGetProjects(username)
      if (res.data.success) {
        projectList.value = res.data.data || []
      }
    } finally {
      loading.value = false
    }
  }

  const createProject = async (projectData: ProjectFormData) => {
    const res = await ApiCreateProject(projectData)
    if (res.data.success) {
      // 重新获取项目列表
      await fetchProjects(projectData.username)
      return res.data.data
    }
    throw new Error(res.data.message)
  }

  return {
    currentProject,
    projectList,
    loading,
    setCurrentProject,
    fetchProjects,
    createProject
  }
})
```

## 5. API接口设计

### 5.1 HTTP请求封装
```typescript
import axios from 'axios'
import { ElMessage } from 'element-plus'
import { useUserStore } from '@/stores/user'

const request = axios.create({
  baseURL: 'http://localhost:9201',
  timeout: 30000
})

// 请求拦截器
request.interceptors.request.use(
  config => {
    const userStore = useUserStore()
    if (userStore.token) {
      config.headers.Authorization = `Bearer ${userStore.token}`
    }
    if (userStore.username) {
      config.headers.username = userStore.username
    }
    return config
  },
  error => Promise.reject(error)
)

// 响应拦截器
request.interceptors.response.use(
  response => response,
  error => {
    if (error.response?.status === 401) {
      const userStore = useUserStore()
      userStore.logout()
      router.push('/login')
    } else {
      ElMessage.error(error.response?.data?.message || '请求失败')
    }
    return Promise.reject(error)
  }
)

export default request
```

### 5.2 API接口定义
```typescript
// 项目相关API
export const ApiGetProjects = (username: string) =>
  request.get('/project-management/list/active/my', {
    headers: { username }
  })

export const ApiCreateProject = (data: ProjectFormData) =>
  request.post('/project-management/create', data)

export const ApiDeleteProject = (data: { projectId: string }) =>
  request.delete('/project-management/delete', { params: data })

export const ApiGetProjectInfo = (projectId: string) =>
  request.get('/project-management/info', { params: { projectId } })

// 配置相关API
export const ApiGetProjectConfig = (projectId: string) =>
  request.get('/project-config/get', { params: { projectId } })

export const ApiSaveProjectConfig = (data: ProjectConfigData) =>
  request.post('/project-config/save', data)

// 文件上传API
export const createUploadUrl = (projectId: string, fileType: string) =>
  `http://localhost:9201/${fileType}/upload?projectId=${projectId}`
```
